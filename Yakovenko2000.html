        # Fraccin aleatoria de dinero a intercambiar

        # Calcular la cantidad de dinero a transferir de i a j
        delta_m = epsilon * (M/N)

        # Asegrese de que el dinero se mantenga no negativo

# Funcin para seleccionar un par de agentes aleatorios y realizar el intercambio
def exchange_money(money, rule_type=3):
     # Elegir dos agentes aleatorios
    i, j = np.random.randint(0, N, 2)

    # Asegurar de que no son el mismo agente
    if i != j:


        # Regla 1: intercambio de una cantidad fija de dinero
        if rule_type == 1:
            delta_m = 1
            if money[i] >= delta_m:
                money[i] -= delta_m
                money[j] += delta_m
        # Regla 2: intercambio de una fraccin aleatoria del dinero promedio
        elif rule_type == 2:
            nu = np.random.random()
            delta_m = nu * (money[i] + money[j]) / 2
            if money[i] >= delta_m:
                money[j] += delta_m
                money[i] -= delta_m
        # Regla 3: intercambio de una fraccin aleatoria del dinero promedio en el sistema
        elif rule_type == 3:
            nu = np.random.random()
            delta_m = nu * M / N
            if money[i] >= delta_m:
                money[i] -= delta_m
                money[j] += delta_m
    return(money)

# Parametros
N =  1000 # Nmero de agentes
M = 10000  # Dinero total en el sistema
iterations = 500001  # Nmero de transacciones

# Inicializar la distribucin de dinero: distribucin equitativa del dinero total entre los agentes
money = np.ones(N) * (M / N)

# Simulacin
for step in range(iterations):
    money = exchange_money(money, rule_type=3)

    # Cada 100000 pasos, se puede visualizar la distribucin
    if step % 100000 == 0:
        plt.clf()
        plt.close()
        plt.figure()
        # Grafique el histograma de la distribucin del dinero
        plt.hist(money, bins=50, density=True, alpha=0.6, color='g', label='Datos simulados')

        #Ajuste a la distribucin de Boltzmann-Gibbs (ajuste exponencial)
        average_money = M / N
        x = np.linspace(0, max(money), 1000)
        boltzmann_gibbs = (1 / average_money) * np.exp(-x / average_money)

        plt.plot(x, boltzmann_gibbs, 'r-', linewidth=2, label='Distribucin Boltzmann-Gibbs')

        plt.xlabel('Dinero por Agente')
        plt.title(f'Distribucin de Dinero en el Paso {step}')
        plt.legend()
        plt.xlim(0, max(money))
        plt.grid(True)
        plt.pause(0.01)

# Mostrar la distribucin final
plt.show()
